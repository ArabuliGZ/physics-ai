\subsection{Производные. Свойства производных}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 
 \z{Вычислите производные:\par% 
	\hpu{$(4\sqrt{x}+1)(2x-3)$}%
	\hpu{$\sqrt{2x^2+1}$}%
	\hpu{$\frac{2-x}{4x^2+3}$}%
	}
%
{\puan$12\sqrt{x}-\frac{6}{\sqrt{x}}+2$;~\puan $\frac{4x}{2\sqrt{2x^2+1}}$;~\puan $\frac{4x^2-16x-3}{(4x^2+3)^2}$}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

 \z{Пусть $x(t)$ некая функция от времени, а все остальные коэффициенты не зависят от времени. Вычислите производные следующих функций по времени:\par%
 	\hpu{$mgx(t)$}% 	
 	\hpu{$\frac{kx^2(t)}{2}$ }%
 	\hpu{$\frac{m(x'(t))^2}{2}$}%
 	}
%
{\puan$mgx'(t)$;~\puan$kx(t)x'(t)$;~\puan$mx'(t)x''(t)$}
	
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

 \z{Вычислите производные (аргумент $x$):\par% 
	\hpu{$\tg{x}$}%
	\hpu{$\sin{(3x^2+1)}$}%
	\hpu{$\sin^2{(3x^2+1)}$}%
	}
%
{\puan$\frac{1}{\cos^2{x}}$;~\puan$6x\cos{(3x^2+1)}$;~\puan$12x\sin{(3x^2+1)}\cos{(3x^2+1)}$}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

 \z{Вычислите значение производных в точке $x_0$:\par%
	\hpu{$3x^2+2,\, x_0=1$}%
	\hpu{$\frac{1}{1+x},\, x_0=1$}%
	\hpu{$3\sin{(x^2-4)},\, x_0=2$}%
	}
%
{\puan$6$;~\puan$-\frac{1}{4}$;~\puan$12$}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

 \z{Найдите точки экстремумов ($a, b$ --- константы, $x$ --- аргумент):\par%
	\hpu{$ax^2+bx+c$}%
	\hpu{$ax+\frac{b}{x}$}%
	\hpu{$a\sin{x}+b\cos{x}$}%
	}
%
{\puan$-\frac{b}{2a}$;~\puan$\pm \sqrt{ab}$;~\puan$\arctg{\frac{a}{b}}$}
