\subsection{Простые интегралы}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 
 \z{Вычислите неопределённые интегралы:\par% 
	\hpu{$\displaystyle \int (2x + 1) \, dx$}%
	\hpu{$\displaystyle \int \sqrt{1 - 2x} \, dx$}%
	}
%
{\puan$x^2 + x + C$;~\puan $-\frac{(1-2x)^{3/2}}{3} + C$}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

 \z{Вычислите неопределённые интегралы:\par%
 	\hpu{$\displaystyle \int x \sqrt{2x^2 - 3} \, dx$}% 	
 	\hpu{$\displaystyle \int \frac{x}{3x^2 - 5} \, dx$ }%
 	}
%
{\puan$\frac{(2x^2-3)^{3/2}}{6} + C$;~\puan$\frac{1}{6} \ln(3x^2-5) + C$}
	
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

 \z{Вычислите определённые интегралы:\par% 
	\hpu{$\displaystyle \int\limits_0^1 x(3x - 1)^2 \, dx$}%
	\hpu{$\displaystyle \int\limits_0^1 \frac{x}{\sqrt{x^2 + 5}} \, dx$}%
	}
%
{\puan$\frac{3}{4}$;~\puan$\sqrt{6} - \sqrt{5}$}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

 \z{Пусть $v(t) = at^2 + bt + c$, вычислите интеграл $\displaystyle \int\limits_0^\tau v(t) \, dt$}
%
{$\frac{a\tau^3}{3} + \frac{b\tau^2}{2} + c\tau$}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

 \z{Пусть $x(t) = 3t^2 + 2$, вычислите интеграл $\displaystyle\int\limits_1^4 \left( x'(t) + 3 \right) dt$}
%
{$54$}
